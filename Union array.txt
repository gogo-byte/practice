#include<iostream>
using namespace std;
int main(){
    int m, n, i = 0, j = 0;
    int arr1[] = {1 , 2, 3, 4, 5, 6};
    int arr2[] = {3, 4, 7, 8, 9};
    m = sizeof(arr1)/sizeof(arr1[0]);
    n = sizeof(arr2)/sizeof(arr2[0]);
    
    while (i < m && j < n){
        
        if (arr1[i] < arr2[j]){
            cout << arr1[i++]<<" ";
        }
        else if (arr2[j] < arr1[i]){
            cout << arr2[j++] << " ";
        }         
        else {
            cout << arr2[j++] << " ";
            i++;
        }


    }  
    while (i < m)
        cout << arr1[i++] << " ";
 
    while (j < n)
        cout << arr2[j++] << " ";
        }
