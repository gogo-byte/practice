#include<iostream>
using namespace std;
int main(){
    int i , j , temp , arr[5] ;
    cout << " Enter 5 array elements ";
    for (i = 0; i < 5; i++){
        cin >> arr[i];
        }
    for (i = 0; i < 5/2; i++){
        j = 4 - i;
        temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
        }
    for (i = 0; i < 5; i++){
        cout << arr[i];
    }
}
